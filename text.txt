CREATE TABLE guests_info(
	g_name VARCHAR(20),
	g_id INTEGER,
	g_birth DATE,
	g_address VARCHAR(20),
	g_email VARCHAR(20),
	g_phone VARCHAR(20),
	PRIMARY KEY(g_id)
);

CREATE TABLE branch_info(
	b_name VARCHAR(20),
	b_id INTEGER,
	b_city VARCHAR(20),
	b_province VARCHAR(20),
	b_country VARCHAR(20),
	PRIMARY KEY(b_id)
);
